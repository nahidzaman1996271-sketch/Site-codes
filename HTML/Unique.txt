    <script>
        // GLOBAL STATE
        const gameState = {
            players: [],
            selectedPlayer: null,
            worldTime: 'dawn',
            divineEnergy: 100,
            eventLog: [],
            playerStats: {
                health: 85,
                faith: 88,
                karma: 72,
                power: 45,
                luck: 60,
                sanity: 90,
                destiny: 50
            }

        // INITIALIZE PLAYERS
        function initializePlayers() {
            const playerNames = ['Aria', 'Kael', 'Lyra', 'Thorne', 'Sariel', 'Vex', 'Orion', 'Mira'];
            const worldMap = document.getElementById('worldMap');
            
            playerNames.forEach((name, index) => {
                const player = {
                    id: index,
                    name: name,
                    x: Math.random() * 85 + 5,
                    y: Math.random() * 85 + 5,
                    color: `hsl(${index * 45}, 70%, 60%)`,
                    stats: {
                        health: Math.floor(Math.random() * 50) + 50,
                        faith: Math.floor(Math.random() * 100),
                        karma: Math.floor(Math.random() * 100) - 50,
                        power: Math.floor(Math.random() * 100),
                        luck: Math.floor(Math.random() * 100),
                        sanity: Math.floor(Math.random() * 50) + 50,
                        destiny: Math.floor(Math.random() * 100)
                    },
                    status: ['Alive', 'Blessed', 'Cursed', 'Enlightened'][Math.floor(Math.random() * 4)],
                    location: ['Forest', 'Desert', 'Mountains', 'City', 'Underworld'][Math.floor(Math.random() * 5)]
                };
                
                gameState.players.push(player);
                
                // Create player dot on map
                const dot = document.createElement('div');
                dot.className = 'player-dot';
                dot.style.left = `${player.x}%`;
                dot.style.top = `${player.y}%`;
                dot.style.backgroundColor = player.color;
                dot.style.boxShadow = `0 0 10px ${player.color}`;
                dot.title = `${player.name} ‚Ä¢ ${player.status}`;
                dot.onclick = () => selectPlayer(player);
                
                worldMap.appendChild(dot);
            });
            
            // Select first player by default
            if (gameState.players.length > 0) {
                selectPlayer(gameState.players[0]);
            }
        }

        // SELECT PLAYER FUNCTION
        function selectPlayer(player) {
            gameState.selectedPlayer = player;
            
            // Update player info display
            document.getElementById('selectedPlayerInfo').innerHTML = `
                <div style="margin-top: 10px;">
                    <strong style="color: ${player.color}">${player.name}</strong><br>
                    Status: ${player.status}<br>
                    Location: ${player.location}<br>
                    Divine Connection: ${player.stats.faith > 70 ? '‚ú¶‚ú¶‚ú¶' : player.stats.faith > 40 ? '‚ú¶‚ú¶' : '‚ú¶'}
                </div>
            `;
            
            // Update game world view
            document.getElementById('currentAvatar').textContent = player.name.charAt(0);
            document.getElementById('currentPlayerName').textContent = player.name;
            document.getElementById('currentPlayerStatus').textContent = 
                `Status: ${player.status} ‚Ä¢ Location: ${player.location} of ${['Whispers', 'Echoes', 'Lost Souls', 'Divinity'][Math.floor(Math.random() * 4)]}`;
            
            // Generate stats controls
            const statsControl = document.getElementById('statsControl');
            statsControl.style.display = 'grid';
            statsControl.innerHTML = '';
            
            Object.entries(player.stats).forEach(([stat, value]) => {
                const controlDiv = document.createElement('div');
                controlDiv.className = 'stat-control';
                controlDiv.innerHTML = `
                    <div class="stat-label">
                        <span>${stat.toUpperCase()}</span>
                        <span class="stat-value" id="statValue-${stat}">${value}</span>
                    </div>
                    <input type="range" min="0" max="100" value="${value}" 
                           class="stat-slider" id="slider-${stat}"
                           oninput="updateStat('${stat}', this.value)"
                           style="background: linear-gradient(to right, #0a0a2a, ${player.color})">
                `;
                statsControl.appendChild(controlDiv);
            });
            
            // Add divine energy tracker
            const energyDiv = document.createElement('div');
            energyDiv.className = 'stat-control';
            energyDiv.innerHTML = `
                <div class="stat-label">
                    <span>DIVINE ENERGY</span>
                    <span class="stat-value" id="divineEnergy">${gameState.divineEnergy}</span>
                </div>
                <input type="range" min="0" max="100" value="${gameState.divineEnergy}" 
                       class="stat-slider" id="slider-energy"
                       oninput="updateDivineEnergy(this.value)"
                       style="background: linear-gradient(to right, #0a0a2a, #ffaa00)">
            `;
            statsControl.appendChild(energyDiv);
            
            logEvent(`${player.name} has been selected by divine will.`, 'world');
        }

        // UPDATE PLAYER STAT
        function updateStat(stat, value) {
            if (!gameState.selectedPlayer) return;
            
            gameState.selectedPlayer.stats[stat] = parseInt(value);
            document.getElementById(`statValue-${stat}`).textContent = value;
            
            // Visual feedback
            const eventLog = document.getElementById('eventLog');
            const logEntry = document.createElement('div');
            logEntry.className = 'log-entry log-miracles';
            logEntry.textContent = `‚ö° ${gameState.selectedPlayer.name}'s ${stat} adjusted to ${value}`;
            eventLog.prepend(logEntry);
            
            // Keep log manageable
            if (eventLog.children.length > 10) {
                eventLog.removeChild(eventLog.lastChild);
            }
        }

        // UPDATE DIVINE ENERGY
        function updateDivineEnergy(value) {
            gameState.divineEnergy = parseInt(value);
            document.getElementById('divineEnergy').textContent = value;
            document.getElementById('divineAttention').innerHTML = `
                <div style="color: ${value > 70 ? '#4aff4a' : value > 30 ? '#ffff4a' : '#ff4a4a'}">
                    ‚ú¶ Divine Attention: ${value > 70 ? 'HIGH' : value > 30 ? 'MODERATE' : 'LOW'}
                </div>
                <div style="font-size: 0.9em;">Energy: ${value}% ‚Ä¢ Connected to ${gameState.players.length} souls</div>
            `;
        }

        // DIVINE POWERS
        function performMiracle() {
            if (gameState.divineEnergy < 10) {
                alert("Insufficient divine energy!");
                return;
            }
            
            gameState.divineEnergy -= 10;
            updateDivineEnergy(gameState.divineEnergy);
            
            const miracles = [
                "A healing light descends from the heavens!",
                "The barren land suddenly blooms with flowers!",
                "A mystical creature appears to guide the lost!",
                "Ancient knowledge floods the seeker's mind!",
                "A protective aura shields the faithful!"
            ];
            
            const miracle = miracles[Math.floor(Math.random() * miracles.length)];
            logEvent(`‚ú® MIRACLE: ${miracle}`, 'miracles');
            
            if (gameState.selectedPlayer) {
                gameState.selectedPlayer.stats.health = Math.min(100, gameState.selectedPlayer.stats.health + 20);
                gameState.selectedPlayer.stats.faith = Math.min(100, gameState.selectedPlayer.stats.faith + 15);
                updateStat('health', gameState.selectedPlayer.stats.health);
                updateStat('faith', gameState.selectedPlayer.stats.faith);
            }
        }

        function performCurse() {
            if (gameState.divineEnergy < 15) {
                alert("Insufficient divine energy!");
                return;
            }
            
            gameState.divineEnergy -= 15;
            updateDivineEnergy(gameState.divineEnergy);
            
            const curses = [
                "A shadow plague infects the land!",
                "Nightmares haunt the unworthy!",
                "Weapons turn against their masters!",
                "The earth rejects your footsteps!",
                "Voices of doubt whisper in your mind!"
            ];
            
            const curse = curses[Math.floor(Math.random() * curses.length)];
            logEvent(`‚ò†Ô∏è CURSE: ${curse}`, 'curses');
            
            if (gameState.selectedPlayer) {
                gameState.selectedPlayer.status = 'Cursed';
                gameState.selectedPlayer.stats.luck = Math.max(0, gameState.selectedPlayer.stats.luck - 25);
                gameState.selectedPlayer.stats.sanity = Math.max(0, gameState.selectedPlayer.stats.sanity - 20);
                updateStat('luck', gameState.selectedPlayer.stats.luck);
                updateStat('sanity', gameState.selectedPlayer.stats.sanity);
            }
        }

        function sendVision() {
            const visions = [
                "You see a future where you rule as a benevolent king/queen!",
                "A prophetic dream reveals hidden treasure in the Crystal Caves!",
                "You witness your own death in three different timelines!",
                "Ancient spirits show you the location of a forgotten temple!",
                "You foresee a great calamity that only you can prevent!"
            ];
            
            const vision = visions[Math.floor(Math.random() * visions.length)];
            logEvent(`üîÆ VISION: ${vision}`, 'miracles');
            
            if (gameState.selectedPlayer) {
                gameState.selectedPlayer.stats.destiny = Math.min(100, gameState.selectedPlayer.stats.destiny + 10);
                updateStat('destiny', gameState.selectedPlayer.stats.destiny);
            }
        }

        function createNPC() {
            const npcTypes = ['Wise Elder', 'Mysterious Stranger', 'Rebel Leader', 'Fallen Angel', 'Ancient Spirit'];
            const npc = npcTypes[Math.floor(Math.random() * npcTypes.length)];
            logEvent(`üë§ Created NPC: ${npc} joins the world`, 'world');
        }

        function modifyPhysics() {
            const modifications = [
                "Gravity reversed in the Floating Isles!",
                "Time flows backward near the Chrono Wells!",
                "Water now burns with eternal flame!",
                "Shadows have become solid matter!",
                "Sound creates visible ripples in reality!"
            ];
            
            const mod = modifications[Math.floor(Math.random() * modifications.length)];
            logEvent(`‚öñÔ∏è PHYSICS MODIFIED: ${mod}`, 'world');
        }

        function controlTime() {
            const timeEffects = ['Time accelerated!', 'Time frozen!', 'Time reversed locally!', 'Day became night instantly!'];
            const effect = timeEffects[Math.floor(Math.random() * timeEffects.length)];
            logEvent(`‚è≥ TIME CONTROL: ${effect}`, 'world');
        }

        function rewriteMemory() {
            const memories = [
                "You remember a childhood that never happened...",
                "The face of your beloved shifts and changes...",
                "That betrayal you witnessed? It was an illusion...",
                "You recall speaking a language you never learned...",
                "The memory of your death surfaces, though you live..."
            ];
            
            const memory = memories[Math.floor(Math.random() * memories.length)];
            logEvent(`üß† MEMORY REWRITTEN: ${memory}`, 'curses');
            
            if (gameState.selectedPlayer) {
                gameState.selectedPlayer.stats.sanity = Math.max(0, gameState.selectedPlayer.stats.sanity - 15);
                updateStat('sanity', gameState.selectedPlayer.stats.sanity);
            }
        }

        function overrideDeath() {
            logEvent(`üíÄ DEATH OVERRIDDEN: A soul is pulled back from the afterlife!`, 'miracles');
            
            if (gameState.selectedPlayer) {
                gameState.selectedPlayer.status = 'Reborn';
                gameState.selectedPlayer.stats.health = 100;
                gameState.selectedPlayer.stats.destiny += 30;
                updateStat('health', 100);
                updateStat('destiny', gameState.selectedPlayer.stats.destiny);
            }
        }

        // PLAYER ACTIONS
        function playerAction(action) {
            if (!gameState.selectedPlayer) return;
            
            const actions = {
                pray: {
                    text: "üôè You pray with deep devotion...",
                    effects: { faith: 10, karma: 5 },
                    logClass: 'player'
                },
                explore: {
                    text: "üß≠ You venture into the unknown...",
                    effects: { luck: 5, power: 5 },
                    logClass: 'player'
                },
                challenge: {
                    text: "‚öîÔ∏è You challenge fate itself!",
                    effects: { power: 15, health: -10, destiny: 10 },
                    logClass: 'player'
                },
                defy: {
                    text: "üë∫ You defiantly challenge divine will!",
                    effects: { faith: -20, power: 20, karma: -15 },
                    logClass: 'player'
                }
            };
            
            const act = actions[action];
            logEvent(act.text, act.logClass);
            
            // Apply effects
            Object.entries(act.effects).forEach(([stat, change]) => {
                if (gameState.selectedPlayer.stats[stat] !== undefined) {
                    gameState.selectedPlayer.stats[stat] = Math.max(0, Math.min(100, 
                        gameState.selectedPlayer.stats[stat] + change));
                    updateStat(stat, gameState.selectedPlayer.stats[stat]);
                }
            });
            
            // Random divine response (20% chance)
            if (Math.random() < 0.2) {
                setTimeout(() => {
                    const responses = [
                        "The heavens seem to acknowledge your action...",
                        "A strange energy pulses through the world...",
                        "Whispers echo from beyond reality...",
                        "Reality shimmers around you momentarily..."
                    ];
                    logEvent(responses[Math.floor(Math.random() * responses.length)], 'world');
                }, 1000);
            }
        }

        // LOG EVENT FUNCTION
        function logEvent(text, type = 'world') {
            const eventLog = document.getElementById('eventLog');
            const logEntry = document.createElement('div');
            logEntry.className = `log-entry log-${type}`;
            logEntry.textContent = text;
            
            eventLog.prepend(logEntry);
            
            // Keep log manageable
            if (eventLog.children.length > 8) {
                eventLog.removeChild(eventLog.lastChild);
            }
            
            // Add to game state
            gameState.eventLog.unshift({ text, type, timestamp: new Date() });
        }

        // AUTO WORLD EVENTS
        function generateWorldEvents() {
            const events = [
                { text: "üåô The twin moons align, creating tidal waves of magic...", type: 'world' },
                { text: "‚ö° A thunderstorm of pure energy approaches from the east...", type: 'world' },
                { text: "üå± The Great Tree sheds leaves that grant visions...", type: 'world' },
                { text: "üë• Pilgrims gather at the Temple of Lost Voices...", type: 'player' },
                { text: "üíÄ A necropolis rises from the desert sands...", type: 'world' }
            ];
            
            if (Math.random() < 0.3) {
                const event = events[Math.floor(Math.random() * events.length)];
                logEvent(event.text, event.type);
            }
            
            // Random player actions
